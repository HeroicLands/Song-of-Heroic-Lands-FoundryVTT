Interface BeingData<TLogic>

Persisted data model for a Being actor. Carries no fields of its own beyond the common SohlActorData base.

Remarks

The shape of system on a being actor — i.e. actor.system (equivalently actor.logic.data) when actor.type === "being". The backing DataModel implements this interface.

interface BeingData<
    TLogic extends SohlActorLogic<BeingData> = SohlActorLogic<any>,
> {
    actionDefs: sohl.entity.action.SohlAction.Data[];
    actorLogic: null | SohlActorLogic<any>;
    appearance: HTMLString;
    dossier: HTMLString;
    hasPlayerOwner: boolean;
    id: string;
    isOwner: boolean;
    itemLogics: SohlItemLogic<any>[];
    kind: string;
    logic: TLogic;
    name: string;
    parent: null | SohlActor;
    portrait: FilePath;
    shortcode: string;
    type: string;

    getFlag(scope: string, key: string): unknown;
    label(options?: { withName: boolean }): string;
    setFlag(scope: string, key: string, value: unknown): Promise<unknown>;
    update(data: object): Promise<unknown>;
}
